    function qualify (value) {
        assert(value != null)
        if (value.prototype instanceof Error || value === Error) {
            return 'error'
        }
        if (value === String) {
            return 'String'
        }
        if (value === Number) {
            return 'Number'
        }
        if (value === Boolean) {
            return 'Boolean'
        }
        if (value instanceof RegExp) {
            return 'regex'
        }
        const type = typeof value
        switch (type) {
        case 'number':
            if (Number.isInteger(value) && value >= 0) {
                return 'range'
            }
            return type
        case 'string':
        case 'object':
            if (Array.isArray(value)) {
                return 'array'
            }
            return type
        }
    }

    const descend = [ 'error', 'array', 'Number', 'Boolean', 'String' ]
    const expected = {
        error: [ 'range', 'object', 'string', 'regex' ],
        array: [ 'array' ],
        number: [ 'number', 'range' ],
        bool: [ 'bool' ],
        string: [ 'string' ],
        bigint: [ 'bigint' ]
    }

    function range (range, object, value) {
        switch (range) {
        case 'max': {
                object.dive[1] = value
                return 'min'
            }
        case 'min': {
                object.dive[0] = object.dive[1]
                object.dive[1] = value
                return 'done'
            }
        case 'done': {
                throw new Error('unexpected range')
            }
        }
    }

    function createObject (type) {
        return { dive: [ 0, Number.MAX_SAFE_INTEGER ], test: { type }, next: null }
    }

    function parse (pattern, parser) {
        while (pattern.length != 0) {
            const shifted = pattern.shift()
            const value = shifted === Infinity ? Number.MAX_SAFE_INTEGER : shifted
            const type = qualify(value)
            if (~descend.indexOf(type)) {
                return parser.done(parse(pattern, parsers[type](createObject(type), type, value)))
            } else {
                assert(~expected[parser.mode].indexOf(type))
                parser.consume(type, value)
            }
        }
        return parser.done()
    }

        function assume (pattern) {
        const options = { match: -1 }
        if (Array.isArray(pattern[0])) {
            if (!pattern[0].every(element => Array.isArray(element))) {
                const vargs = pattern.shift()
                assert(vargs.length == 1 && typeof vargs[0] == 'number')
                options.match = vargs[0]
            }
        }
        const type = qualify(pattern[0])
        if (~descend.indexOf(type)) {
            return {
                ...options,
                dive: [ 0, 0 ],
                test: { type: 'root' },
                next: parse(pattern, parsers[type](createObject(type), type, pattern.shift()))
            }
        }
        return {
            ...options,
            dive: [ 0, 0 ],
            test: { type: 'root' },
            next: parse(pattern, parsers.error(createObject('error'), 'error', Error))
        }
    }

    assert(pattern != null && typeof pattern == 'object')

    return assume(pattern.slice())
